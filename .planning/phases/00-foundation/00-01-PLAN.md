For now, implement as:
    ```typescript
    import logger from '@/lib/logger';

    export async function logAuditEvent(
      model: string,
      operation: string,
      args: unknown,
      result: unknown,
      userId?: string,
    ): Promise<void> {
      // Import basePrisma lazily to avoid circular deps
      const { basePrisma } = await import('@/lib/db');
      try {
        await basePrisma.auditLog.create({
          data: {
            model,
            record_id: extractRecordId(result) ?? 'unknown',
            operation,
            before: operation === 'update' ? (args as any)?.where ?? null : null,
            after: result ? JSON.parse(JSON.stringify(result)) : null,
            user_id: userId ?? 'system',
          },
        });
      } catch (error) {
        logger.error({ error, model, operation }, 'Failed to write audit log');
      }
    }

    function extractRecordId(result: unknown): string | null {
      if (result && typeof result === 'object' && 'id' in result) {
        return String((result as { id: unknown }).id);
      }
      return null;
    }
    ```
